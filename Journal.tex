\newcommand{\stepSize}{0.5}

\newcommand{\ruled}{
    %I believe this is a much better way to rule the paper
    \begin{tikzpicture}
        \foreach \y in {0,\stepSize,...,\tikzHeight}
        \draw[gray, very thin] (0,\y) -- (\tikzWidth,\y);
    \end{tikzpicture}
}

\newcommand{\bullets}{
    \begin{tikzpicture}
            % setup the nodes
            \foreach \x in {0,\stepSize,...,\tikzWidth}
            \foreach \y in {0,\stepSize,...,\tikzHeight}{
                \node[mystyle] at (\x,\y){};
            }
    \end{tikzpicture}
}

\newcommand{\grid}{
    \begin{tikzpicture}
        \draw[step=\stepSize,gray,very thin] (0,0) grid (\tikzWidth, \tikzHeight);
    \end{tikzpicture}
}
